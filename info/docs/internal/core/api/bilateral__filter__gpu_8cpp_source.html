19// Structure matching the WGSL Uniform (std140 layout)
20struct FilterParams {
21 float sigmaSpatial;
22 float sigmaRange;
23 // std140 requires 16-byte alignment for structs/vec4s.
24 // Two floats = 8 bytes. We likely need padding if this struct grows,
25 // but for a standalone bind group of 2 floats, standard alignment usually suffices.
26 // However, it is safer to pad to 16 bytes to be sure.
27 float _pad1;
28 float _pad2;
29} __attribute__((packed));

31void bilateral_filter_gpu(uint8_t* image, size_t width, size_t height, double sigma_spatial,
32 double sigma_range, uint8_t color_space) {
33 if (sigma_spatial <= 0.0 || sigma_range <= 0.0 || width <= 0 || height <= 0) return;
34 if (color_space != COLOR_SPACE_OPTION_CIELAB && color_space != COLOR_SPACE_OPTION_RGB) return;
35
36 std::vector<uint8_t> result(width * height * 4);
37 // copy image data to result incase filter fails
38 std::memcpy(result.data(), image, width * height * 4);
39 // CIELAB conversion will run as shader
40
41 std::cout << "begin wgpu portion" << std::endl;
42 // 1. Create Input Texture
43 wgpu::TextureDescriptor texDesc = {};
44 texDesc.size = {static_cast<uint32_t>(width), static_cast<uint32_t>(height), 1};
45 int bytesPerPixel{4};
46 texDesc.format = wgpu::TextureFormat::RGBA8Unorm;
47 texDesc.usage = wgpu::TextureUsage::TextureBinding | wgpu::TextureUsage::CopyDst;
48 wgpu::Texture inputTexture = GPU::getClassInstance().get_device().CreateTexture(&texDesc);
49
50 std::cout << "upload texture" << std::endl;
51 // Upload data to Input Texture
52 wgpu::TexelCopyTextureInfo dst = {};
53 dst.texture = inputTexture;
54 wgpu::TexelCopyBufferLayout layout = {};
55 layout.offset = 0;
56
57 layout.bytesPerRow = width * bytesPerPixel; // Tightly packed for upload
58 layout.rowsPerImage = height;
59 GPU::getClassInstance().get_queue().WriteTexture(&dst, image, bytesPerPixel * width * height,
60 &layout, &texDesc.size);
61
62 std::cout << "create output texture" << std::endl;
63 // 2. Create Output Texture (Storage)
64 wgpu::TextureDescriptor outDesc = texDesc;
65 outDesc.usage = wgpu::TextureUsage::StorageBinding | wgpu::TextureUsage::CopySrc;
66 wgpu::Texture outputTexture = GPU::getClassInstance().get_device().CreateTexture(&outDesc);
67
68 // 2a. Intermediate textures for LAB if needed
69 wgpu::TextureDescriptor descLab = texDesc;
70 descLab.format = wgpu::TextureFormat::RGBA32Float; // <--- CRITICAL
71 descLab.usage = wgpu::TextureUsage::StorageBinding | wgpu::TextureUsage::TextureBinding;
72 // input lab
73 wgpu::Texture texLabRaw = GPU::getClassInstance().get_device().CreateTexture(&descLab);
74 // filtered lab
75 wgpu::Texture texLabFiltered = GPU::getClassInstance().get_device().CreateTexture(&descLab);
76
77 std::cout << "create buffer" << std::endl;
78 // 3. Create Uniform Buffer
79 float sr = static_cast<float>(sigma_range);
80 FilterParams params = {static_cast<float>(sigma_spatial), sr, 0.0f, 0.0f};
81 wgpu::BufferDescriptor bufDesc = {};
82 bufDesc.size = sizeof(FilterParams);
83 bufDesc.usage = wgpu::BufferUsage::Uniform | wgpu::BufferUsage::CopyDst;
84 wgpu::Buffer paramBuffer = GPU::getClassInstance().get_device().CreateBuffer(&bufDesc);
85 GPU::getClassInstance().get_queue().WriteBuffer(paramBuffer, 0, &params, sizeof(FilterParams));
86
87 wgpu::ComputePipeline pipeline;
88 wgpu::ComputePipeline pipelineRGB2LAB;
89 wgpu::ComputePipeline pipelineLAB2RGB;
90
91 switch (color_space) {
92 case COLOR_SPACE_OPTION_RGB: {
93 pipeline = GPU::getClassInstance().createPipeline("bilateral_filter_rgb",
94 "BilateralFilterShader");
95 break;
96 }
97 case COLOR_SPACE_OPTION_CIELAB: {
98 // also requires RGB-CIELAB conversion shaders
99 pipeline = GPU::getClassInstance().createPipeline("bilateral_filter_lab",
100 "BilateralFilterShader");
101 pipelineRGB2LAB = GPU::getClassInstance().createPipeline("rgb2cielab", "rgb2lab");
102 pipelineLAB2RGB = GPU::getClassInstance().createPipeline("cielab2rgb", "lab2rgb");
103 break;
104 }
105 }
106
107 // 6. Create Bind Group
108
109 // filter bind group
110 wgpu::BindGroupDescriptor bindGroupDesc = {};
111 bindGroupDesc.layout = pipeline.GetBindGroupLayout(0);
112 wgpu::BindGroupEntry entries[3];
113 // Entry 0: Input Texture View
114 entries[0].binding = 0;
115 entries[1].binding = 1;
116
117 switch (color_space) {
118 case COLOR_SPACE_OPTION_RGB: {
119 entries[0].textureView = inputTexture.CreateView();
120 // Entry 1: Output Texture View
121 entries[1].textureView = outputTexture.CreateView();
122 break;
123 }
124 case COLOR_SPACE_OPTION_CIELAB: {
125 entries[0].textureView = texLabRaw.CreateView();
126 // Entry 1: Output Texture View
127 entries[1].textureView = texLabFiltered.CreateView();
128 break;
129 }
130 }
131 // Entry 2: Uniform Buffer
132 entries[2].binding = 2;
133 entries[2].buffer = paramBuffer;
134 entries[2].size = sizeof(FilterParams);
135 bindGroupDesc.entryCount = 3;
136 bindGroupDesc.entries = entries;
137 wgpu::BindGroup bindGroup =
138 GPU::getClassInstance().get_device().CreateBindGroup(&bindGroupDesc);
139
140 wgpu::BindGroup bindGroupRGB2LAB;
141 wgpu::BindGroup bindGroupLAB2RGB;
142
143 if (color_space == COLOR_SPACE_OPTION_CIELAB) {
144 // rgb2lab bind group
145 wgpu::BindGroupEntry bg1Entries[2];
146 bg1Entries[0].binding = 0;
147 bg1Entries[0].textureView = inputTexture.CreateView();
148 bg1Entries[1].binding = 1;
149 bg1Entries[1].textureView = texLabRaw.CreateView();
150 wgpu::BindGroupDescriptor bg1Desc = {};
151 bg1Desc.layout = pipelineRGB2LAB.GetBindGroupLayout(0);
152 bg1Desc.entryCount = 2;
153 bg1Desc.entries = bg1Entries;
154 bindGroupRGB2LAB = GPU::getClassInstance().get_device().CreateBindGroup(&bg1Desc);
155 // lab2rgb bind group
156 wgpu::BindGroupEntry bg2Entries[2];
157 bg2Entries[0].binding = 0;
158 bg2Entries[0].textureView = texLabFiltered.CreateView();
159 bg2Entries[1].binding = 1;
160 bg2Entries[1].textureView = outputTexture.CreateView();
161 wgpu::BindGroupDescriptor bg2Desc = {};
162 bg2Desc.layout = pipelineLAB2RGB.GetBindGroupLayout(0);
163 bg2Desc.entryCount = 2;
164 bg2Desc.entries = bg2Entries;
165 bindGroupLAB2RGB = GPU::getClassInstance().get_device().CreateBindGroup(&bg2Desc);
166 }
167
168 // 7. Dispatch Compute Pass
169 wgpu::CommandEncoder encoder = GPU::getClassInstance().get_device().CreateCommandEncoder();
170
171 if (color_space == COLOR_SPACE_OPTION_CIELAB) {
172 wgpu::ComputePassEncoder pass1 = encoder.BeginComputePass();
173 pass1.SetPipeline(pipelineRGB2LAB);
174 pass1.SetBindGroup(0, bindGroupRGB2LAB);
175 // Workgroups of 16x16
176 pass1.DispatchWorkgroups((width + 15) / 16, (height + 15) / 16);
177 pass1.End();
178 }
179
180 wgpu::ComputePassEncoder pass = encoder.BeginComputePass();
181 pass.SetPipeline(pipeline);
182 pass.SetBindGroup(0, bindGroup);
183 // Workgroups of 16x16
184 pass.DispatchWorkgroups((width + 15) / 16, (height + 15) / 16);
185 pass.End();
186
187 if (color_space == COLOR_SPACE_OPTION_CIELAB) {
188 wgpu::ComputePassEncoder pass2 = encoder.BeginComputePass();
189 pass2.SetPipeline(pipelineLAB2RGB);
190 pass2.SetBindGroup(0, bindGroupLAB2RGB);
191 // Workgroups of 16x16
192 pass2.DispatchWorkgroups((width + 15) / 16, (height + 15) / 16);
193 pass2.End();
194 }
195
196 // 8. Prepare for Readback (Copy Texture -> Buffer)
197 // We cannot read textures directly on CPU. We must copy to a MapRead buffer.
198 uint32_t alignedBytesPerRow =
199 GPU::getAlignedBytesPerRow(width, static_cast<uint32_t>(bytesPerPixel));
200 uint32_t bufferSize = alignedBytesPerRow * height;
201
202 wgpu::BufferDescriptor readBufDesc = {};
203 readBufDesc.size = bufferSize;
204 readBufDesc.usage = wgpu::BufferUsage::MapRead | wgpu::BufferUsage::CopyDst;
205 wgpu::Buffer readBuffer = GPU::getClassInstance().get_device().CreateBuffer(&readBufDesc);
206
207 wgpu::TexelCopyTextureInfo srcTex = {};
208 srcTex.texture = outputTexture;
209
210 wgpu::TexelCopyBufferInfo dstBuf = {};
211 dstBuf.buffer = readBuffer;
212 dstBuf.layout.bytesPerRow = alignedBytesPerRow;
213
214 encoder.CopyTextureToBuffer(&srcTex, &dstBuf, &texDesc.size);
215
216 wgpu::CommandBuffer commands = encoder.Finish();
217 GPU::getClassInstance().get_queue().Submit(1, &commands);
218 std::cout << "queue submit" << std::endl;

222 // 9. Map Async (To read data back to C++)
223 // In a real app, you likely pass a callback function here.
224 struct ReadbackContext {
225 wgpu::Buffer buffer;
226 uint32_t size;
227 uint32_t alignedBytesPerRow;
228 int width;
229 int height;
230 };
231
232 uint8_t* result_ptr = result.data();
233 bool* waiting = new bool(true);
234
235 readBuffer.MapAsync(
236 wgpu::MapMode::Read, 0, bufferSize, wgpu::CallbackMode::AllowProcessEvents,
237 [](wgpu::MapAsyncStatus status, wgpu::StringView message, void* userdata) {
238 std::cout << "In callback" << std::endl;
239 bool* flag = static_cast<bool*>(userdata);
240 bool success = false;
241 if (status == wgpu::MapAsyncStatus::Success) {
242 success = true;
243 // std::cout << "Map success: " << message.data << std::endl;
244 } else {
245 // Handle error
246 success = false;
247 // std::cerr << "Map failed: " << message.data << std::endl;
248 }
249 *flag = false;
250 },
251 (void*)waiting);
252
253 std::cout << "waiting " << *waiting << std::endl;
254
255 while (*waiting) {
256 GPU::getClassInstance().get_instance().ProcessEvents();
257#if defined(__EMSCRIPTEN__)
258 emscripten_sleep(10);
259#endif
260 }
261 std::cout << "done wgpu" << std::endl;
262 const uint8_t* mappedData = (const uint8_t*)readBuffer.GetConstMappedRange(0, bufferSize);
263 // copy to cpu buffer
264 for (size_t y = 0; y < height; ++y) {
265 const uint8_t* rowPtr = mappedData + (y * alignedBytesPerRow);
266 for (size_t x = 0; x < width; ++x) {
267 const uint8_t* pixelPtr = rowPtr + (x * bytesPerPixel);
268 size_t dstIndex = 4 * (y * width + x); // RGBA
269
270 std::memcpy(&result_ptr[dstIndex], pixelPtr, sizeof(uint8_t));
271 std::memcpy(&result_ptr[dstIndex + 1], pixelPtr + 1, sizeof(uint8_t));
272 std::memcpy(&result_ptr[dstIndex + 2], pixelPtr + 2, sizeof(uint8_t));
273 std::memcpy(&result_ptr[dstIndex + 3], pixelPtr + 3, sizeof(uint8_t));
274 }
275 }
276 readBuffer.Unmap();
277 std::memcpy(image, result.data(), result.size());
278 std::cout << "done memcpy" << std::endl;
279
280 // explicit clean up
281
282 if (inputTexture) inputTexture.Destroy();
283 if (outputTexture) outputTexture.Destroy();
284 if (texLabRaw) texLabRaw.Destroy();
285 if (texLabFiltered) texLabFiltered.Destroy();
286 readBuffer.Destroy();
287 delete waiting;
288#if defined(__EMSCRIPTEN__)
289 emscripten_sleep(50);
290#endif
291 result.clear();
292 result.shrink_to_fit();
293}
